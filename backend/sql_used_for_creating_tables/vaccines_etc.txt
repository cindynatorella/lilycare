CREATE TABLE vaccines (
	id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
	name TEXT NOT NULL,
	description TEXT NOT NULL DEFAULT '',
	note TEXT NOT NULL DEFAULT '',
	recurrence_months INTEGER NOT NULL CHECK (recurrence_months > 0),
	next_due DATE
);

CREATE TABLE vaccine_history_entries (
	id INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
	vaccine_id INTEGER NOT NULL REFERENCES vaccines(id) ON DELETE CASCADE,
	administered_on DATE NOT NULL,
	UNIQUE (vaccine_id, administered_on)
);

CREATE INDEX idx_vaccine_history_entries_vaccine_id
ON vaccine_history_entries(vaccine_id);
